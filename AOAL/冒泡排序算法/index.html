Array.prototype.sort = function(){
	var i, j, temp, len = this.length;
	for(i = 0; i < len - 1; i++){
		for(j = 0; j < len - i - 1; j++){
			if(this[j] > this[j + 1]){
				temp = this[j];
				this[j] = this[j + 1];
				this[j + 1] = temp;
			}
		}
	}
	return this;
}
