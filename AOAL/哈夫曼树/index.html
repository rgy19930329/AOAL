算法思路

给定n个权值作为n的叶子结点，构造一棵二叉树，若带权路径长度达到最小，称这样的二叉树为最优二叉树，也称为哈夫曼树(Huffman Tree)。

为了使得到的哈夫曼树的结构尽量唯一，通常规定生成的哈夫曼树中每个结点的左子树根结点的权小于等于右子树根结点的权。

哈夫曼树的建立步骤：对权值进行从小到大的排序，每次选择剩余权值中最小值与次小值构建局部的哈夫曼树，所得的父节点的值为二者之和。若该值不小于接下来的最小值和次小值，就进行分支生长；否则继续生长。

function BiNode(data){
	this.left = null;
	this.right = null;
	this.data = data;
}

function createHuffman(arr){
	var arr = arr.concat();
	arr.sort(function(a, b){
		return a - b;
	});
	var tarr = [];

	for(var i = 0; i < arr.length; i++){
		var node = new BiNode(arr[i]);
		tarr[i] = node;
	}

	for(var i = 0; i < arr.length - 1; i++){
		var temp = tarr.splice(0, 2);
		var data = temp[0].data + temp[1].data;
		var node = new BiNode(data);
		node.left = temp[0];
		node.right = temp[1];

		tarr.push(node);
		tarr.sort(function(node1, node2){
			return node1.data - node2.data;
		});
	}

	return tarr[0];
}
