Array.prototype.quickSort = function(){
	if(this.length <= 1){
        return this;
    }else{
    	var leftArray = [],
		rightArray = [],
		index = Math.floor(this.length / 2),
		middleValue = this.splice(index, 1);

		leftArray = this.filter(function(item){
			return item <= middleValue;
		});

		rightArray = this.filter(function(item){
			return item > middleValue;
		});

		return leftArray.quickSort().concat(middleValue, rightArray.quickSort());
    }
}
