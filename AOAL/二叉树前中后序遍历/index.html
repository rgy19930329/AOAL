算法思路

所谓遍历(Traversal)是指沿着某条搜索路线，依次对二叉树中每个节点均做一次且仅做一次访问。

二叉树的前中后序遍历有两种不同的思路，一种是用递归思想，另一种是通过队列进行临时存储从而达到目的。这里采用的是递归方法实现。

前序遍历：根-左-右

中序遍历：左-根-右

后序遍历：左-右-根

// 二叉树节点
function BiNode(){
	this.left = null;
	this.right = null;
	this.data = null;
}

// 按层建立二叉树
function createBiTree(arr){

	var root = new BiNode();
	var tarr = [];
	var front  = 1, rear = 0;
	while(rear < arr.length){

		var node = new BiNode();
		node.data = arr[rear];
		node.left = null;
		node.right = null;
		rear++;
		tarr[rear] = node;

		if(rear == 1){
			root = node;
		}else{
			var front = Math.floor( rear / 2);
			if(rear % 2 == 0){
				tarr[front].left = node;
			}else{
				tarr[front].right = node;
				front++;
			}
		}
	}
	return root;
}
